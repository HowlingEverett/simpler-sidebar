How to install this sidebar:

 $( document ).ready(function() {
    $( '#sidebar' ).simplerSidebar({
        opener: '#toggle-sidebar',
        attr: 'simplersidebar',
        animation: {
            easing: 'easeOutQuint'
        },
        sidebar: {
            width: 300,
            closingLinks: '.close-sb',
        }
    });
});
