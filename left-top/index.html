How to install this sidebar:

 $( document ).ready(function() {
    $( '#sidebar' ).simplerSidebar({
        opener: '#toggle-sidebar',
        dataName: 'simplersidebar',
        animation: {
            easing: 'easeOutQuint'
        },
        sidebar: {
            align: 'left',
            width: 300,
            closingLinks: '.close-sb',
        }
    });
});
